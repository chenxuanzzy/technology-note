# 目標

當我點擊按鈕時會動態切換圖片

# 步驟拆解

先抓到 img 的 DOM ，在加上 img 的 src

let img = document.querySelector('img')  //抓到 DOM
img.src = 'cat.jpg' // 放入你要的圖片

# 範例

照慣例請搭配簡單範例 (opens new window)

  <div class="container mt-5">
    <div class="row">
    <p>我是貓派</p>
    <img src="https://static.independent.co.uk/2021/06/16/08/newFile-4.jpg?width=982&height=726&auto=webp&quality=75" alt="lover_cat">
    <button class="btn bg-warning mt-3">狗派點我</button>
    </div>
  </div>

const api = 'https://dog.ceo/api/breeds/image/random'
let img = document.querySelector('img')
let btn = document.querySelector('button')
console.log(btn)
btn.addEventListener('click',(event) =>{
  axios.get(api)
    .then(data => {
      console.log(data)
      img.src = data.data.message
      //將 img.src 切換成 回傳 的圖片網址
  }).catch(error =>{
      console.log(error)
  })
})
